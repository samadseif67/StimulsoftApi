<script>


    function PaymentMellt() {
        this.Pay = (amount) => {
            console.log(`amount is ${amount} by Mellat`);
        }
    }

    function PaymentSaman() {

        this.Pay = (amount) => {
            console.log(`amount is ${amount} by saman`);
        }
    }


    function PaymentSaderat() {
        this.Pay = (amount) => {
            console.log(`amount is ${amount} is saderat`);
        }
    }


    function PaymentStertegy() {

        this.Pay = (type, amount) => {
            switch (type) {
                case 'Mellat': return new PaymentMellt().Pay(amount); break;
                case 'Saman': return new PaymentSaman().Pay(amount); break;
                case 'Saderat': return new PaymentSaderat().Pay(amount); break;
            }
        }
    }
