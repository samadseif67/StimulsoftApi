اطلاعات خروجی یک پروسیجر 

SELECT rs.column_ordinal, rs.name,rs.system_type_name
FROM    sys.dm_exec_describe_first_result_set
        ('EXEC dbo.spLoadGridContracts', NULL, 0) AS rs
JOIN    sys.types AS t
        ON t.user_type_id = rs.system_type_id

		order by rs.column_ordinal asc
