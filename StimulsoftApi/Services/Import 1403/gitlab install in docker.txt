ایجاد فایل docker-compose.yml

----------------------------------------------------------

version: "3"

services:
  gitlab:
    image: gitlab/gitlab-ce:latest
    container_name: gitlab
    restart: always
    hostname: "localhost"
    ports:
      - "8020:80"
      - "1443:443"
      - "2222:22"
    volumes:
      - ./config:/etc/gitlab
      - ./logs:/var/log/gitlab
      - ./data:/var/opt/gitlab




-----------------------------------------------
docker compose up -d
