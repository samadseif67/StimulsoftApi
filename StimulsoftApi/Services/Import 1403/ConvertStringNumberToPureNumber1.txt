   public string ConvertStringNumberToPureNumber(string Number = "")
   {
       string Result = "";

       for (int i = 0; i < Number.Length; i++)
       {
           if (IsNumberCharacter(Number[i]))
           {
               Result += Number[i];
           }
       }
       return Result;
   }
